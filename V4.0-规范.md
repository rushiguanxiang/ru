## 核心铁律（永不更改）

| 项目 | 定义 |
|------|------|
| 华夏数理 | 从1开始，不用0 |
| 0x00 | 保留/空/未定义/异常入口 |
| 本象 | 0x01-0x40（64卦，序号1-64） |
| 易象 | 0x41-0x80（64卦，序号65-128） |
| 易态标记 | 「乄」（U+4E44），永不更改 |
| 机器码结构 | [7]=恒零, [6:1]=六爻（上→初）, [0]=乄位 |
| 纳甲法则 | 内卦定体（目标寄存器），外卦定用（操作类型） |

### 1.2 五行权限法则

```
金(0)：R-- 或 R-X（只读/执行，固化不可改）
木(1)：RW-（可读写，动态生长）
水(2)：RW-（流式FIFO，只进不出或环形）
火(3)：RW-（高速缓存，回写/写通）
土(4)：RW-（静态直写，段基址）
```

### 1.3 五音宽度法则

```
宫(0)：8位  商(1)：16位  角(2)：32位  徵(3)：64位  羽(4)：128位
```

### 1.4 CAM条目结构（32位）

```
[31:27] = 纳音编码（5位，0-29）
[26]    = 环选择（0=周天轮，1=甲子轮）
[25:19] = 基址（7位）
[18:13] = 长度（6位，0-63字）
[12:10] = 权限（3位：R/W/X）
[9:7]   = 五行（3位）
[6:4]   = 五音（3位）
[3:2]   = 策略（2位：0直写,1回写,2写通）
[1]     = 环形标志
[0]     = 有效位
```

## 卷三：太极Hello World

### 3.1 目标

在甲子轮（大林木/堆区）生成斐波那契数列前8项，通过午通道（UART/炉中火）输出。

### 3.2 完整可运行代码

```cpp
#include <iostream>
#include <array>
#include <cstdint>

// ========================
// 《如语言 V4.0》核心仿真头
// ========================
class WanchuanCore {
public:
    // 十天干寄存器（甲~癸）
    std::array<uint8_t, 10> reg{0};

    // 双环存储
    std::array<uint8_t, 128> zhouTianRing{0};
    std::array<uint8_t, 60>  jiaZiRing{0};

    // 纳音MMU条目
    struct NayinEntry {
        uint8_t  nayinCode : 5;
        uint8_t  ringSel   : 1;
        uint8_t  base      : 7;
        uint8_t  length    : 6;
        uint8_t  perm      : 3;
        uint8_t  xing      : 3;
        uint8_t  yin       : 3;
        uint8_t  strategy  : 2;
        bool     isRing    : 1;
        bool     valid     : 1;
    };
    std::array<NayinEntry, 30> nayinMMU;

    // 四柱锁（当前场域字）
    uint32_t fieldWord = 0;

    // 状态
    uint8_t  pc = 0;
    uint8_t  ringPtr = 0;
    bool     xiheMode = false;

    WanchuanCore() {
        // 初始化纳音MMU
        nayinMMU[0]  = {0,  0, 0x00, 32, 0x5, 0, 0, 0, false, true};
        nayinMMU[1]  = {1,  1, 0x20, 16, 0x7, 3, 1, 1, true,  true};
        nayinMMU[2]  = {2,  1, 0x30, 48, 0x7, 1, 2, 0, true,  true};
        // ...其余条目按需填充
    }

    // 场域字解析
    struct Field {
        uint8_t rippleLvl : 2;
        bool    ringSel   : 1;
        uint8_t reserved  : 2;
        uint8_t dizhi     : 4;
        uint8_t nayin     : 5;
        uint8_t tiangan   : 4;
        uint8_t jiaziPh   : 2;
    };

    Field decodeField(uint32_t fw) {
        return {
            static_cast<uint8_t>((fw >> 18) & 0x3),
            static_cast<bool>((fw >> 17) & 0x1),
            static_cast<uint8_t>((fw >> 15) & 0x3),
            static_cast<uint8_t>((fw >> 11) & 0xF),
            static_cast<uint8_t>((fw >> 6) & 0x1F),
            static_cast<uint8_t>((fw >> 2) & 0xF),
            static_cast<uint8_t>(fw & 0x3)
        };
    }

    // 纳甲提取
    uint8_t getInnerTrigram(uint8_t hex) { return (hex >> 1) & 0x7; }
    uint8_t getOuterTrigram(uint8_t hex) { return (hex >> 4) & 0x7; }

    uint8_t trigramToGan(uint8_t tri) {
        switch (tri) {
            case 0b111: return 0; case 0b000: return 1;
            case 0b001: return 2; case 0b110: return 3;
            case 0b010: return 4; case 0b101: return 5;
            case 0b100: return 6; case 0b011: return 7;
            default: return 0;
        }
    }

    uint8_t trigramToOp(uint8_t tri) {
        switch (tri) {
            case 0b111: return 8; case 0b000: return 9;
            case 0b001: return 2; case 0b110: return 3;
            case 0b010: return 4; case 0b101: return 5;
            case 0b100: return 6; case 0b011: return 7;
            default: return 0;
        }
    }

    // 执行单条指令
    void execute(uint8_t hexagram, uint32_t field) {
        Field f = decodeField(field);
        uint8_t inner = getInnerTrigram(hexagram);
        uint8_t outer = getOuterTrigram(hexagram);
        bool yi = hexagram & 0x1;

        uint8_t targetGan = trigramToGan(inner);
        uint8_t opType = trigramToOp(outer);

        if (!yi) {
            std::cout << "  [本态预备] 外=" << (int)outer << " 内=" << (int)inner << std::endl;
            pc = (pc + 1) & 0x7F;
            return;
        }

        std::cout << "  [易态执行] 操作=" << (int)opType << " 目标=" << (int)targetGan;

        switch (opType) {
            case 4: { // 坎戊算术
                if (hexagram == 0x2B || hexagram == 0x67) { // 既济乄
                    reg[targetGan] = reg[0] + reg[1];
                    std::cout << " 既济:甲+乙=" << (int)reg[targetGan];
                } else if (hexagram == 0x55 || hexagram == 0x68) { // 未济乄
                    reg[targetGan] = reg[0] - reg[1];
                    std::cout << " 未济:甲-乙=" << (int)reg[targetGan];
                }
                break;
            }
            case 5: { // 离己逻辑
                if (hexagram == 0x4B || hexagram == 0x66) { // 贲乄
                    reg[targetGan] = reg[0] | reg[2];
                    std::cout << " 贲:甲|丙=" << (int)reg[targetGan];
                } else if (hexagram == 0x53 || hexagram == 0x63) { // 噬嗑乄
                    reg[targetGan] = reg[0] & reg[2];
                    std::cout << " 噬嗑:甲&丙=" << (int)reg[targetGan];
                }
                break;
            }
            case 3: { // 兑丁端口
                if (hexagram == 0x35 || hexagram == 0x75) { // 归妹乄
                    std::cout << " [UART输出] " << (int)reg[targetGan];
                }
                break;
            }
        }
        std::cout << std::endl;

        // 涟漪处理
        switch (f.rippleLvl) {
            case 0: pc = (pc + 1) & 0x7F; break;
            case 1: {
                uint8_t adj = (ringPtr + 1) % 60;
                jiaZiRing[adj] = reg[targetGan];
                pc = (pc + 1) & 0x7F;
                break;
            }
            case 2: {
                pc = 0; ringPtr = 0;
                for (auto& e : nayinMMU) if (e.valid) e.perm = 0x7;
                break;
            }
        }
    }

    void loadProgram(const std::vector<uint8_t>& code, uint8_t start = 0) {
        for (size_t i = 0; i < code.size() && start + i < 128; i++)
            zhouTianRing[start + i] = code[i];
        pc = start;
    }

    void run(uint32_t defaultField) {
        fieldWord = defaultField;
        while (pc < 128 && zhouTianRing[pc] != 0) {
            uint8_t hex = zhouTianRing[pc];
            std::cout << "PC=" << (int)pc << " 象码0x" << std::hex << (int)hex << std::dec;
            execute(hex, fieldWord);
        }
    }
};

// ========================
// 太极Hello World
// ========================
int main() {
    WanchuanCore cpu;

    // 设置纳音MMU
    cpu.nayinMMU[2]  = {2, 1, 0x30, 8, 0x7, 1, 2, 0, true, true};
    cpu.nayinMMU[1]  = {1, 1, 0x20, 4, 0x7, 3, 1, 1, true, true};

    // 初始化斐波那契
    cpu.reg[0] = 0;   // 甲 = F(0)
    cpu.reg[1] = 1;   // 乙 = F(1)
    cpu.reg[2] = 0;   // 丙 = 临时
    cpu.ringPtr = 0;

    std::cout << "=== 太极Hello World：斐波那契 ===" << std::endl;

    // 场域字：大林木，1级涟漪，目标丙
    uint32_t field = (0 << 18) | (1 << 17) | (2 << 11) | (2 << 6) | (2 << 2);

    // 生成8项，利用甲子轮天然循环
    for (int i = 0; i < 8; i++) {
        // 既济乄：丙 = 甲 + 乙
        cpu.execute(0x2B, field);
        cpu.jiaZiRing[cpu.ringPtr] = cpu.reg[2];
        std::cout << "F(" << i << ")=" << (int)cpu.reg[2] << " ";

        // 未济乄：甲 = 乙
        cpu.execute(0x55, field);
        cpu.reg[0] = cpu.reg[1];

        // 既济乄：乙 = 丙
        cpu.execute(0x2B, field);
        cpu.reg[1] = cpu.reg[2];

        cpu.ringPtr = (cpu.ringPtr + 1) % 60;
    }
    std::cout << std::endl;

    // UART输出
    uint32_t uart_field = (0 << 18) | (1 << 17) | (6 << 11) | (1 << 6) | (3 << 2);
    std::cout << "=== UART输出 ===" << std::endl;
    for (int i = 0; i < 8; i++) {
        cpu.reg[3] = cpu.jiaZiRing[i];
        cpu.execute(0x75, uart_field);
    }
    std::cout << std::endl;

    return 0;
}
```
